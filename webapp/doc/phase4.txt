CREATE TABLE planets (
    id SERIAL,
    pl_hostname TEXT,
    pl_discmethod TEXT,
    pl_pnum INT,
    pl_orbper INT,
    pl_orbsmax INT,
    pl_orbeccen INT,
    pl_massj INT,
    pl_bmassprov TEXT,
    pl_radj INT,
    pl_dens INT,
    pl_ttvflag TEXT,
    pl_kepflag TEXT,
    pl_k2flag TEXT,
    pl_nnotes INT
    pl_facility TEXT,
    row_update TEXT
);

CREATE TABLE stars (
    id SERIAL,
    st_name TEXT,
    st_pnum INT,
    st_planet_1_name TEXT,
    st_planet_2_name TEXT,
    st_planet_3_name TEXT,
    st_planet_4_name TEXT,
    st_planet_5_name TEXT,
    st_planet_6_name TEXT,
    st_planet_7_name TEXT,
    st_planet_8_name TEXT,
    st_dist INT,
    st_teff INT,
    st_mass INT,
    st_rad INT,
    row_update TEXT
);

QUERY (GET): /planets/<pl_name>/
RESPONSE: A dictionary that contains all available data for the planet as a key
EXAMPLE: http://whatever.com/planets/11ComB/
